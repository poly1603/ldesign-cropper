🚀 快速开始

npm install @ldesign/cropper

import { Cropper } from '@ldesign/cropper'
import '@ldesign/cropper/dist/style.css'

const cropper = new Cropper('#container', {
  src: 'image.jpg',
  aspectRatio: 16 / 9,
  keyboard: true,  // 启用快捷键
  filters: true    // 启用滤镜
})
